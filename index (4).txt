        function save() {
            localStorage.setItem('supreme_hist_v6', JSON.stringify(history));
            localStorage.setItem('supreme_stats_v6', JSON.stringify(stats));

        // --- CORE ENGINE ---
        async function engineLoop() {
            try {
                const res = await fetch(API_URL + "&ts=" + Date.now());
                const json = await res.json();
                const list = json.data.list;
                const latestIssue = list[0].issueNumber;
                const latestRes = parseInt(list[0].number);
                const latestSize = latestRes >= 5 ? 'BIG' : 'SMALL';
                const latestColor = [1,3,7,9].includes(latestRes) ? 'GREEN' : (latestRes === 5 ? 'GREEN' : 'RED');

                // 1. UPDATE LIVE METERS & GRID
                liveGrid = list.slice(0, 6).map(i => parseInt(i.number));
                calculateMeters(list);

                // 2. RESOLVE PENDING
                if(history.length > 0 && history[0].status === 'analyzing') {
                    if(history[0].period === latestIssue) {
                        const win = history[0].pred === (history[0].type === 'SIZE' ? latestSize : latestColor);
                        history[0].status = win ? 'win' : 'loss';
                        history[0].result = `${latestRes}`;
                        
                        if(win) { stats.win++; if(history[0].nums.includes(latestRes)) stats.jackpot++; }
                        else stats.loss++;
                        
                        save();
                        renderView();
                    }
                }

                // 3. GENERATE NEW PREDICTION
                const nextPeriod = (BigInt(latestIssue) + 1n).toString();
                if(lastFetchedPeriod !== latestIssue) {
                    lastFetchedPeriod = latestIssue;

                    // ANTI-CHOP LOGIC
                    let sCount = 0; let cCount = 0;
                    for(let i=0; i<10; i++) {
                        let n = parseInt(list[i].number);
                        if((n >= 5 ? 'BIG':'SMALL') === latestSize) sCount++;
                        let c = [1,3,7,9].includes(n) ? 'GREEN' : (n===5?'GREEN':'RED');
                        if(c === latestColor) cCount++;
                    }

                    let pSize = ""; let pType = "SIZE";
                    let algo = "AI SNIPER V4";

                    // Check ZigZag (Volatility)
                    let isZigZag = (list[0].number >= 5) !== (list[1].number >= 5);
                    
                    if(cCount >= sCount && !isZigZag) {
                        pSize = latestColor; pType = 'COLOR'; algo = "COLOR FLOW";
                    } else {
                        // If zig-zag, flip trend
                        if(isZigZag) { pSize = latestSize === 'BIG' ? 'SMALL' : 'BIG'; algo = "ANTI-CHOP"; }
                        else { pSize = latestSize; algo = "TREND FOLLOW"; }
                    }

                    const n1 = (latestRes + 3) % 10;
                    const n2 = (latestRes + 7) % 10;

                    currentPred = { period: nextPeriod, p: pSize, n: [n1, n2], method: algo };

                    history.unshift({
                        period: nextPeriod, method: algo, pred: pSize, type: pType,
                        nums: [n1, n2], status: 'analyzing', result: null
                    });
                    if(history.length > 25) history.pop();
                    
                    save();
                    renderView();
                }

            } catch(e) {}
        }

        function calculateMeters(list) {
            // Volatility
            let switches = 0;
            for(let i=0; i<9; i++) {
                if((parseInt(list[i].number)>=5) !== (parseInt(list[i+1].number)>=5)) switches++;
            }
            meters.vol = Math.min(100, switches * 12);

            // Strength
            let streak = 1;
            let type = parseInt(list[0].number) >= 5;
            for(let i=1; i<10; i++) {
                if((parseInt(list[i].number)>=5) === type) streak++; else break;
            }
            meters.str = Math.min(100, streak * 20);
        }
